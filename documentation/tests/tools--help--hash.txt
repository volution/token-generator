>>  z-tokens hash --help

Usage:
  z-tokens hash [OPTIONS]


Optional arguments:
  -h,--help             Show this help message and exit
  -a,--algorithm {algorithm}
                        (hashing algorithm)
  --md5                 (use MD5)
  --sha1                (use SHA1)
  --sha2                (use SHA2)
  --sha3                (use SHA3)
  --blake2              (use Blake2s or Blake2b)
  --blake2s             (use Blake2s)
  --blake2b             (use Blake2b)
  --blake3              (use Blake3)
  --git-sha1            (use Git SHA1 flavour)
  --argon2              (use one of Argon2 family) !!! EXPERIMENTAL !!!
  --argon2d             (use Argon2d) !!! EXPERIMENTAL !!!
  --argon2i             (use Argon2i) !!! EXPERIMENTAL !!!
  --argon2id            (use Argon2id) !!! EXPERIMENTAL !!!
  --siphash             (use SipHash) !!! WEAK !!!
  --seahash             (use SeaHash) !!! WEAK !!!
  --highway             (use HighwayHash) !!! WEAK !!!
  --xxhash              (use xxHash) !!! WEAK !!!
  --xxh3                (use Xxh3) !!! WEAK and EXPERIMENTAL !!!
  --djb2                (use djb2) !!! WEAK !!!
  --sdbm                (use SDBM) !!! WEAK !!!
  --fnv1a               (use FNV1a) !!! WEAK !!!
  --crc                 (use CRC) !!! WEAK !!!
  --crc32c              (use CRC32C) !!! WEAK !!!
  --adler               (use Adler) !!! WEAK !!!
  -s,--size {size}      (hash output size in bytes)
  --8b,--1B             (output 8 bits / 1 byte)
  --16b,--2B            (output 16 bits / 2 bytes)
  --32b,--4B            (output 32 bits / 4 bytes)
  --64b,--8B            (output 64 bits / 8 bytes)
  --128b,--16B          (output 128 bits / 16 bytes)
  --256b,--32B          (output 256 bits / 32 bytes)
  --512b,--64B          (output 512 bits / 64 bytes)
  --1024b,--128B        (output 1024 bits / 128 bytes)
  --2048b,--256B        (output 2048 bits / 256 bytes)
  --4096b,--512B        (output 4096 bits / 512 bytes)
  --8192b,--1024B       (output 8192 bits / 1024 bytes)
  -i,--stdin            (read from stdin)
  -f,--file {input}     (read from file)
  -t,--token {input}    (use this argument)
